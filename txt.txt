// 1. Read saved form data on first load
const [form, setForm] = useState(() => {
  const saved = localStorage.getItem('registerForm');
  return saved ? JSON.parse(saved) : { name: '', email: '', password: '', confirm: '' };
});

// 2. Save form to localStorage whenever it changes
useEffect(() => {
  localStorage.setItem('registerForm', JSON.stringify(form));
}, [form]);
